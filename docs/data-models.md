## 1. 核心数据类型（TypeScript）

### 1.1 LyricLine（歌词行）

```typescript
interface LyricLine {
  /** 唯一标识符，创建时自动生成 */
  id: string;                          // UUID v4

  /** 行号，从0开始，用于排序 */
  index: number;

  /** 该行的所有字符/词单元 */
  characters: LyricChar[];

  /** 该行的开始时间（毫秒）。等于第一个char的startTime。
   *  未打轴时为null */
  startTime: number | null;

  /** 该行的结束时间（毫秒）。等于最后一个char的endTime。
   *  未打轴时为null */
  endTime: number | null;

  /** 该行是否被标记为"跳过"（如间奏行） */
  isSkipped: boolean;

  /** 行标签（如：'verse', 'chorus', 'bridge'），可选 */
  tag: string | null;

  /** 行级别样式覆盖，不设置则使用全局 */
  styleOverride: Partial<LyricStyle> | null;

  /** 注音模式覆盖，不设置则使用全局 */
  pronunciationMode: PronunciationMode | null;
}
```

### 1.2 LyricChar（歌词字符/词单元）

```typescript
interface LyricChar {
  /** 唯一标识符 */
  id: string;                          // UUID v4

  /** 父行ID */
  lineId: string;

  /** 在该行中的索引位置，从0开始 */
  index: number;

  /** 显示的文本内容。
   *  中文：单个汉字或标点
   *  日文：单个假名或汉字
   *  英文：一个单词或音节 */
  text: string;

  /** 该单元的开始时间（毫秒），未打轴时为null */
  startTime: number | null;

  /** 该单元的结束时间（毫秒），未打轴时为null */
  endTime: number | null;

  /** 持续时间（毫秒），endTime - startTime 的缓存值，方便计算 */
  get duration(): number | null {
    if (this.startTime === null || this.endTime === null) return null;
    return this.endTime - this.startTime;
  }

  /** 是否为标点符号（标点默认附加到前一个单元，不单独打轴） */
  isPunctuation: boolean;

  /** 是否为空格/分隔符（不参与打轴） */
  isSpace: boolean;

  /** 扫色动画的缓动曲线 */
  easing: EasingType;

  /** 填充方向覆盖 */
  fillDirection: FillDirection | null;

  /** 注音数据 */
  pronunciation: Pronunciation | null;

  /** 字级别样式覆盖 */
  styleOverride: Partial<CharStyle> | null;

  /** 是否为手工分字（true=用户手动调整过分词边界） */
  isManualSplit: boolean;
}
```

### 1.3 Pronunciation（注音数据）

```typescript
interface Pronunciation {
  /** 注音模式 */
  mode: PronunciationMode;

  /** 注音文本（如拼音'wǒ'、假名'わたし'）。
   *  对于多音字，存储实际使用的读音 */
  text: string;

  /** 声调/重音标记（拼音用：1-5，1=阴平...5=轻声） */
  tone: number | null;

  /** 注音是否由系统自动生成 */
  isAutoGenerated: boolean;

  /** 注音的备用读音列表（多音字的其他可能读音） */
  alternatives: string[];
}

type PronunciationMode =
  | 'none'       // 不显示注音
  | 'pinyin'     // 汉语拼音
  | 'furigana'   // 日文假名
  | 'romaji'     // 日文罗马字
  | 'bopomofo'   // 注音符号
  | 'hangul'     // 韩文谚文
  | 'custom';    // 自定义
```

### 1.4 样式数据

```typescript
interface LyricStyle {
  /** 样式名称（用于预设） */
  name: string;

  /** 字体设置 */
  font: FontConfig;

  /** 未演唱时的颜色 */
  unsungColor: RGBA;

  /** 已演唱时的颜色 */
  sungColor: RGBA;

  /** 扫色动画设置 */
  sweep: SweepConfig;

  /** 描边设置 */
  outline: OutlineConfig;

  /** 阴影设置 */
  shadow: ShadowConfig;

  /** 发光设置 */
  glow: GlowConfig;

  /** 背景条设置 */
  backgroundBar: BackgroundBarConfig;

  /** 布局设置 */
  layout: LayoutConfig;

  /** 注音样式 */
  pronunciationStyle: PronunciationStyleConfig;
}

interface FontConfig {
  family: string;            // 字体族名，如 'Arial', '思源黑体'
  size: number;              // 字号，像素，范围12-200
  weight: number;            // 字重 100-900 (400=normal, 700=bold)
  italic: boolean;           // 斜体
  letterSpacing: number;     // 字间距，像素，范围0-50
  lineSpacing: number;       // 行间距，像素，范围0-100
}

interface RGBA {
  r: number;  // 0-255
  g: number;  // 0-255
  b: number;  // 0-255
  a: number;  // 0-1
}

interface SweepConfig {
  /** 填充方向 */
  direction: FillDirection;

  /** 渐变过渡带宽度（像素），0 = 硬边界 */
  gradientWidth: number;

  /** 过渡带类型 */
  gradientType: 'linear' | 'ease-in' | 'ease-out' | 'smooth-step';

  /** 默认缓动曲线 */
  defaultEasing: EasingType;

  /** 是否启用已唱文字的发光效果 */
  sungGlow: boolean;

  /** 扫色时是否带动画扩散（如已唱部分文字微微放大） */
  animateScale: boolean;
  scaleAmount: number;       // 缩放量 (1.0 = 不变, 1.05 = 放大5%)
}

type FillDirection =
  | 'left-to-right'
  | 'right-to-left'
  | 'top-to-bottom'
  | 'bottom-to-top';

type EasingType =
  | 'linear'
  | 'ease-in'
  | 'ease-out'
  | 'ease-in-out';

interface OutlineConfig {
  enabled: boolean;
  width: number;             // 像素，0-20
  color: RGBA;
  style: 'outer' | 'inner' | 'center';  // 外描边/内描边/居中
}

interface ShadowConfig {
  enabled: boolean;
  offsetX: number;           // 像素，-20 ~ 20
  offsetY: number;           // 像素，-20 ~ 20
  blur: number;              // 像素，0-30
  color: RGBA;
}

interface GlowConfig {
  enabled: boolean;
  radius: number;            // 像素，0-50
  color: RGBA;
  intensity: number;         // 0-1
}

interface BackgroundBarConfig {
  enabled: boolean;
  color: RGBA;
  /** 背景条高度模式 */
  heightMode: 'auto' | 'fixed';
  /** 固定模式下的高度（像素） */
  fixedHeight: number;
  /** 高度内边距（像素），自适应模式下的额外上下空间 */
  paddingVertical: number;
  /** 水平边距（像素），背景条超出文字宽度多少 */
  paddingHorizontal: number;
  /** 圆角半径（像素） */
  borderRadius: number;
  /** 宽度模式 */
  widthMode: 'text' | 'fullscreen' | 'fixed';
  /** 固定模式下的宽度（像素） */
  fixedWidth: number;
}

interface LayoutConfig {
  /** 水平对齐 */
  horizontalAlign: 'left' | 'center' | 'right';

  /** 垂直对齐 */
  verticalAlign: 'top' | 'center' | 'bottom';

  /** 自定义垂直位置（当verticalAlign不适用时）。
   *  值=距顶部的百分比，如50=居中。null=使用verticalAlign */
  customYPercent: number | null;

  /** 文字排列方向 */
  textDirection: 'horizontal' | 'vertical';

  /** 显示行数模式 */
  lineDisplayMode: 'single' | 'double' | 'all';

  /** 安全区边距（像素） */
  safeAreaMargin: number;
}

interface PronunciationStyleConfig {
  /** 注音字号比例（相对于正文字号） */
  fontSizeRatio: number;          // 默认0.45

  /** 注音与正文间距比例 */
  spacingRatio: number;           // 默认0.1

  /** 注音颜色 */
  color: RGBA;

  /** 注音是否参与扫色 */
  participateSweep: boolean;

  /** 注音对齐方式 */
  alignment: 'center' | 'left' | 'right';
}

/** 字级别样式覆盖（仅支持部分属性） */
interface CharStyle {
  unsungColor?: RGBA;
  sungColor?: RGBA;
  fontSize?: number;
  fontWeight?: number;
  italic?: boolean;
}
```

### 1.5 项目元数据

```typescript
interface ProjectMeta {
  /** 项目文件格式版本（用于向后兼容） */
  version: string;                   // 如 "1.0.0"

  /** 项目名称 */
  title: string;

  /** 歌手名字 */
  artist: string;

  /** 创建时间 ISO8601 */
  createdAt: string;

  /** 最后修改时间 ISO8601 */
  modifiedAt: string;

  /** 创建此项目的软件名+版本 */
  createdBy: string;                 // 如 "Karaoke Lyrics Maker v1.0.0"

  /** 画布设置 */
  canvas: {
    width: number;                   // 如 1920
    height: number;                  // 如 1080
    frameRate: number;               // 如 30
  };

  /** 媒体文件引用 */
  media: MediaReference;

  /** 当前选中的样式预设名称 */
  activeStylePreset: string;

  /** 全局注音模式 */
  pronunciationMode: PronunciationMode;
}

interface MediaReference {
  /** 视频文件路径（相对于项目文件的路径，或绝对路径） */
  videoPath: string | null;

  /** 音频文件路径。如果不为null，覆盖视频中的音频轨 */
  audioPath: string | null;

  /** 视频文件信息（导入时记录，用于检测文件变化） */
  videoInfo: MediaFileInfo | null;

  /** 音频文件信息 */
  audioInfo: MediaFileInfo | null;

  /** 音频音量增益（dB） */
  audioGain: number;

  /** 音频淡入时长（毫秒），0=无 */
  fadeInDuration: number;

  /** 音频淡出时长（毫秒），0=无 */
  fadeOutDuration: number;
}

interface MediaFileInfo {
  /** 文件路径（绝对路径，用于唯一标识） */
  absolutePath: string;

  /** 文件大小（字节），用于检测文件是否被替换 */
  fileSize: number;

  /** 文件最后修改时间 ISO8601 */
  lastModified: string;

  /** 文件哈希（SHA-256前16字节），快速检测文件变更 */
  hashShort: string;

  /** 时长（毫秒） */
  duration: number;

  /** 视频信息（仅视频文件） */
  video?: {
    width: number;
    height: number;
    codec: string;              // 如 'h264', 'hevc'
    frameRate: number;          // 如 29.97
    bitRate: number;            // bps
  };

  /** 音频信息 */
  audio?: {
    codec: string;              // 如 'aac', 'mp3'
    sampleRate: number;         // 如 44100, 48000
    channels: number;           // 1=mono, 2=stereo
    bitRate: number;            // bps
  };
}
```

### 1.6 完整的项目数据结构

```typescript
interface Project {
  /** 元数据 */
  meta: ProjectMeta;

  /** 歌词行数组 */
  lines: LyricLine[];

  /** 样式预设集合 */
  stylePresets: StylePreset[];

  /** 时间线标记数据 */
  timeline: TimelineData;

  /** 撤销历史（保存时清空） */
  undoStack: UndoEntry[];

  /** 导出设置预设 */
  exportPresets: ExportPreset[];
}

interface StylePreset {
  id: string;
  name: string;
  style: LyricStyle;
  isBuiltIn: boolean;        // true=内置预设, false=用户创建
}

interface TimelineData {
  /** 用户设置的BPM（用于节拍吸附），null=未设置 */
  bpm: number | null;

  /** 时间标记列表（打轴过程中的原始按键记录，用于调试/恢复） */
  rawMarks: RawMark[];

  /** 用户设置的段落标记 */
  sectionMarks: SectionMark[];
}

interface RawMark {
  /** 标记时间（毫秒），用户按键时的音频时间 */
  time: number;

  /** 标记类型 */
  type: 'char_start' | 'char_end' | 'line_start' | 'line_end';

  /** 目标字符ID（如果是char级标记） */
  targetCharId: string | null;

  /** 目标行ID（如果是line级标记） */
  targetLineId: string | null;

  /** 是否有手动修正过 */
  isAdjusted: boolean;

  /** 记录时间 */
  recordedAt: string;  // ISO8601
}

interface SectionMark {
  /** 段落名称 */
  name: string;          // 如 '前奏', '第一段', '副歌'
  /** 开始时间（毫秒） */
  startTime: number;
  /** 结束时间（毫秒） */
  endTime: number;
  /** 颜色标记（在时间线上显示） */
  color: string;         // HEX颜色
}

interface UndoEntry {
  /** 操作描述 */
  description: string;

  /** 操作类型 */
  type: 'text_edit' | 'timing_edit' | 'style_edit' | 'split_edit';

  /** 序列化的差异数据 */
  patch: unknown;
}

interface ExportPreset {
  id: string;
  name: string;
  config: ExportConfig;
}
```

## 2. 项目文件格式 (.klproj)

### 2.1 文件结构

`.klproj` 文件是一个 **ZIP压缩包**，内部结构如下：

```
MySong.klproj  (ZIP, Deflate压缩)
│
├── project.json                 # 核心项目数据 (Project 接口的JSON序列化)
│
├── media/                        # 媒体文件引用目录
│   └── references.json          # 媒体文件路径映射
│       {
│         "video": {
│           "relativePath": null,       // 如果媒体在项目旁，存相对路径
│           "absolutePath": "/Users/.../song.mp4",
│           "embedded": false,          // 是否嵌入(大文件通常不嵌入)
│           "info": { ... }             // MediaFileInfo
│         },
│         "audio": null
│       }
│
├── fonts/                        # 嵌入字体（可选，用户选择嵌入时使用）
│   ├── CustomFont-Regular.ttf
│   └── CustomFont-Bold.ttf
│
├── thumbnails/                   # 视频缩略图（可选，加速项目打开）
│   ├── thumb_0001.jpg           # 视频第1秒处截图
│   ├── thumb_0002.jpg           # 视频第10秒处截图
│   └── ...
│
└── waveform/                     # 缓存的波形数据（可选，加速项目打开）
    └── waveform.bin              # f32数组的二进制dump
```

### 2.3 版本兼容策略

- 主版本号（major）：不兼容的格式变更
- 次版本号（minor）：向后兼容的新增字段
- 修订号（patch）：Bug修复，格式不变

加载时：
1. 读取`meta.version`
2. 如果major > 当前支持的major → 提示"此项目由更新版本创建，请升级软件"
3. 如果minor > 当前支持的minor → 加载但提示"项目可能包含此版本不支持的功能"
4. 未知字段 → 保留但不处理（保留往返保存能力）

## 3. 导出文件格式

### 3.1 标准LRC格式

```
[ti:歌曲名称]
[ar:歌手名称]
[al:专辑名称]
[by:制作者名称]
[offset:0]
[re:卡拉OK歌词制作软件]

[00:12.34]第一句歌词文本
[00:15.67]第二句歌词文本
[00:18.90]第三句歌词文本
```

规则：
- 时间标签格式：`[mm:ss.xx]`，mm=分钟(2位补零)，ss=秒(2位补零)，xx=百分之一秒(2位补零)
- 支持可选三位毫秒：`[mm:ss.xxx]`
- 一行支持多个时间标签：`[00:12.34][00:45.67]重复歌词`
- 编码：UTF-8（可选用GBK/Shift-JIS）
- 元数据标签可选

### 3.2 增强LRC（逐字时间）

```
[ti:歌曲名]
[ar:歌手名]

[00:12.34]歌<00:12.34>词<00:12.67>制<00:13.01>作<00:13.34>
[00:15.67]软<00:15.67>件<00:16.01>
```

规则：
- `<mm:ss.xx>` 标签标记每个字/词的开始时间
- 最后一个字的结束时间由下一行或行结束时间决定
- 如果字没有单独标签，从前一个字时间均匀分配

### 3.5 ASS字幕格式

```
[Script Info]
; Generated by Karaoke Lyrics Maker v1.0.0
Title: 歌曲名称
ScriptType: v4.00+
WrapStyle: 0
ScaledBorderAndShadow: yes
PlayResX: 1920
PlayResY: 1080

[V4+ Styles]
Format: Name, Fontname, Fontsize, PrimaryColour, SecondaryColour, OutlineColour, BackColour, Bold, Italic, Underline, StrikeOut, ScaleX, ScaleY, Spacing, Angle, BorderStyle, Outline, Shadow, Alignment, MarginL, MarginR, MarginV, Encoding
Style: Default,Microsoft YaHei,48,&H00FFFFFF,&H000000FF,&H00000000,&H00000000,0,0,0,0,100,100,2,0,1,2,2,2,10,10,10,1

[Events]
Format: Layer, Start, End, Style, Name, MarginL, MarginR, MarginV, Effect, Text
Dialogue: 0,0:00:12.34,0:00:15.67,Default,,0,0,0,,{\kf$123}歌{\kf$134}词{\kf$130}制{\kf$139}作
```

规则：
- `\kf` 标签后跟的 `$` 后数字是**百分之一秒**的持续时间
- 如 `{\kf$123}` 表示该字持续123厘秒 = 1.23秒
- \kf 是"karaoke fill"的缩写，表示从字开始后多久完成填充

## 4. 导入文件格式解析

### 4.1 LRC文件解析规则

```
输入示例:
[ti:Hello]
[ar:Adele]
[00:12.34]Hello, it's me
[00:15.67]I was wondering if after all these years
[00:18.90]<00:18.90>You'd<00:19.20> like<00:19.60> to<00:20.00> meet

解析输出:
- 元数据: { title: "Hello", artist: "Adele" }
- 行:
  - { text: "Hello, it's me", startTime: 12340, chars: [] }
  - { text: "I was wondering if after all these years", startTime: 15670, chars: [] }
  - { text: "You'd like to meet", startTime: 18900,
      chars: [
        { text: "You'd", startTime: 18900 },
        { text: " like", startTime: 19200 },
        { text: " to", startTime: 19600 },
        { text: " meet", startTime: 20000 }
      ]
    }
```

解析器状态机：
```
START → 读取一行
  ├── 空行 → SKIP
  ├── [ti:...] → 设置title
  ├── [ar:...] → 设置artist
  ├── [al:...] → 设置album
  ├── [by:...] → 设置author
  ├── [offset:...] → 设置全局偏移
  ├── [mm:ss.xx] 文本 → 新歌词行，行级时间
  ├── [mm:ss.xx] 文本<mm:ss.xx>文本 → 新歌词行，含逐字时间
  └── 纯文本 → 新歌词行，无时间
```

### 4.2 SRT文件解析规则

```
输入示例:
1
00:00:12,340 --> 00:00:15,670
Hello, it's me

2
00:00:15,670 --> 00:00:18,900
I was wondering if after all these years

解析输出:
- 每个字幕块 → 一个LyricLine
- startTime = 开始时间戳 (HH:MM:SS,mmm → 毫秒)
- endTime = 结束时间戳
- text = 文本内容（HTML标签如<i>去除）
- 无逐字时间
```

### 4.3 纯文本导入规则

```
输入: 用户粘贴的原始文本

处理流程:
1. 编码检测 → 统一转UTF-8
2. 按换行符(\n, \r\n, \r)分割
3. 每行trim()
4. 空行处理：
   - 连续空行合并为1个
   - 首尾空行去除
   - 保留的空行转为isSkipped=true的空歌词行（用于段落分隔）
5. 分字处理：
   - 中文: 逐字分割 (含标点)
   - 日文: 按字符分割 (后续可调整为分词)
   - 英文: 按空格分词，标点附加到前一个词
   - 混合: 按语言边界分割
6. 生成LyricLine和LyricChar对象
```

### 4.4 带注音的文本导入

```
输入格式A (方括号):
我[wo3] 爱[ai4] 你[ni3]

输入格式B (花括号):
私{わたし}は 学{がく}生{せい}です

输入格式C (Ruby标记):
<ruby>私<rt>わたし</rt></ruby>

解析输出:
- 每个[...]或{...}中的内容 → Pronunciation.text
- 自动设置pronunciation.mode
- 括号外的文字 → LyricChar.text
```

### 6.3 导出配置类型

```typescript
interface ExportConfig {
  /** 导出类型 */
  type: 'video' | 'lrc' | 'word_json' | 'ass';

  /** 输出路径 */
  outputPath: string;

  /** 导出范围 */
  range: {
    mode: 'all' | 'selection' | 'time_range';
    /** 选中的行ID列表（mode=selection时） */
    lineIds?: string[];
    /** 时间范围毫秒（mode=time_range时） */
    startMs?: number;
    endMs?: number;
  };

  /** 是否包含注音数据 */
  includePronunciation: boolean;

  // === 视频导出专属设置 ===
  videoConfig?: {
    format: 'mp4_h264' | 'mp4_h265' | 'webm_vp9' | 'mov_prores';
    width: number;
    height: number;
    frameRate: number;
    videoBitrateMbps: number;
    audioBitrateKbps: number;
    encoderPreset: 'ultrafast' | 'superfast' | 'veryfast' | 'medium' | 'slow';
    renderingMode: 'ffmpeg_drawtext' | 'frame_sequence' | 'auto';
  };

  // === LRC导出专属设置 ===
  lrcOptions?: {
    includeMetadata: boolean;
    timeFormat: 'hundredths' | 'thousandths';  // [mm:ss.xx] vs [mm:ss.xxx]
    encoding: 'utf-8' | 'utf-16' | 'gbk';
    wordLevel: boolean;  // 是否生成增强LRC（逐字时间标签）
  };

  // === JSON导出专属设置 ===
  jsonOptions?: {
    format: 'full' | 'compact';
    includeMetadata: boolean;
    includeLineTiming: boolean;
    includePronunciation: boolean;
    prettyPrint: boolean;
  };

  // === ASS导出专属设置 ===
  assOptions?: {
    playResX: number;
    playResY: number;
    fontName: string;
    fontSize: number;
  };
}
```
